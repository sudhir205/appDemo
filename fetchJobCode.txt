export const fetchAvailableItems = async (): Promise<Item[]> => {
  const response = await fetch('/api/items/available');
  if (!response.ok) throw new Error('Failed to fetch available items');
  const data = await response.json(); // [{ jobCode: "J001" }]
  return data.map((item: { jobCode: string }) => ({
    id: item.jobCode,
    label: item.jobCode,
  }));
};
